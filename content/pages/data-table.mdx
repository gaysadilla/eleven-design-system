      description: Table with custom column renderers and server-side operations
      language: typescript
      code: |
        import { DataTable, type ColumnDef } from '@eleven/design-system';
        import { Badge } from '@eleven/design-system';

        interface User {
          id: number;
          name: string;
          email: string;
          role: 'admin' | 'user' | 'guest';
          status: 'active' | 'inactive';
          lastLogin: Date;
        }

        const columns: ColumnDef<User>[] = [
          {
            key: 'name',
            header: 'Name',
            sortable: true,
            render: (value, row) => (
              <div className="font-medium">{value}</div>
            )
          },
          {
            key: 'role',
            header: 'Role',
            filterable: true,
            render: (value) => (
              <Badge variant={value === 'admin' ? 'default' : 'secondary'}>
                {value}
              </Badge>
            )
          },
          {
            key: 'status',
            header: 'Status',
            render: (value) => (
              <Badge variant={value === 'active' ? 'success' : 'warning'}>
                {value}
              </Badge>
            )
          },
          {
            key: 'lastLogin',
            header: 'Last Login',
            sortable: true,
            render: (value) => new Date(value).toLocaleDateString()
          }
        ];

        export function UserTable() {
          const [data, setData] = useState<User[]>([]);
          const [loading, setLoading] = useState(false);
          const [selected, setSelected] = useState<User[]>([]);
          
          const handleSort = async (sort: SortConfig) => {
            setLoading(true);
            // Implement server-side sorting
            const sortedData = await fetchUsers({ sort });
            setData(sortedData);
            setLoading(false);
          };
          
          return (
            <DataTable
              data={data}
              columns={columns}
              selectable
              pagination={{ pageSize: 25 }}
              loading={loading}
              onSelectionChange={setSelected}
              onSortChange={handleSort}
            />
          );
        }

        }
